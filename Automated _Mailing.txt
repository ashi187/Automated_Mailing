If your system is a standalone script running locally, you don't "upload" the file; you simply read it from a known location on the computer.
To read a file named file_lamp_automatic.txt in the same directory as your Python script, you would use this code:

def read_configuration_file(filename="file_lamp_automatic.txt"):
    try:
        with open(filename, 'r') as file:
            content = file.read()
            # Process the content for your milling system logic
            print(f"Successfully read file: {filename}")
            # Example: Split content into lines
            lines = content.splitlines()
            return lines
    except FileNotFoundError:
        print(f"Error: The file '{filename}' was not found.")
        return None

file_data = read_configuration_file()

if file_data:
    print("\nFirst few lines of the file:")
    for line in file_data[:5]:
        print(line)

2. For a Web-Based System (Using Flask or Django)
If your project has a web interface (highly likely for an "automated system" that might be monitored remotely), you'll need a web framework to handle the HTTP file upload request. Flask is a very popular choice for simpler projects like this.
Example using Flask:
Install Flask: pip install Flask
Create a simple upload script (app.py):
<!-- end list -->

from flask import Flask, request, redirect, url_for, render_template
import os

app = Flask(__name__)
# Configure the folder where uploaded files will be saved
UPLOAD_FOLDER = 'uploads' 
app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER
# Create the uploads folder if it doesn't exist
os.makedirs(UPLOAD_FOLDER, exist_ok=True) 

@app.route('/', methods=['GET', 'POST'])
def upload_file():
    if request.method == 'POST':
        # Check if the post request has the file part
        if 'file' not in request.files:
            return redirect(request.url)
        file = request.files['file']
        
        # If the user does not select a file, the browser submits an empty part without a filename
        if file.filename == '':
            return redirect(request.url)
            
        if file:
            # Securely save the file to the UPLOAD_FOLDER
            filepath = os.path.join(app.config['UPLOAD_FOLDER'], file.filename)
            file.save(filepath)
            
            # You can now call your milling system function to process this file
            # process_milling_data(filepath) 
            
            return 'File uploaded successfully and saved to: ' + filepath
            
    # Display the upload form on a GET request
    return '''
    <!doctype html>
    <title>Upload New File</title>
    <h1>Upload a File for the Milling System</h1>
    <form method=post enctype=multipart/form-data>
      <input type=file name=file>
      <input type=submit value=Upload>
    </form>
    '''

if __name__ == '__main__':
    # Run the server
    app.run(debug=True)

3. Run the server and access it in your browser (usually http://127.0.0.1:5000/).
